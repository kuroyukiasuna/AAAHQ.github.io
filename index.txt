function loadYulu() {
  var yulutxt = document.getElementById("maincontent");

      yulutxt.innerHTML = this.responseText;
    }
  };
  xhttp.open("GET", "https://raw.githubusercontent.com/kuroyukiasuna/AAA-Association/master/texts/a-kun%20yulu.txt", true);
  xhttp.send();
};
